 /* Insert Patient */

 CREATE Procedure insertpatientprocedure (IN patientnames varchar(255),IN fathernames varchar(255), IN ages varchar(255),IN genders varchar(255),IN telephone1s varchar(255),IN telephone2s varchar(255), IN mrnos varchar(255))


BEGIN

IF NOT EXISTS (select patientid from patient where mr_no = mrnos) THEN


insert into patient(patientname,fathername,age,gender,telephone1,telephone2,mr_no) values (patientnames,fathernames,ages,genders,telephone1s,telephone2s,mrnos);


ELSE

SIGNAL SQLSTATE '45000'
SET MESSAGE_TEXT = 'Patient MR Number Exist';

END IF;

 END
