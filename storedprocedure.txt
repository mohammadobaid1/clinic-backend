 /* Insert Patient */

 CREATE Procedure insertpatientprocedure (IN patientnames varchar(255),IN fathernames varchar(255), IN ages varchar(255),IN genders varchar(255),IN telephone1s varchar(255),IN telephone2s varchar(255), IN mrnos varchar(255))


BEGIN

IF NOT EXISTS (select patientid from patient where mr_no = mrnos) THEN


insert into patient(patientname,fathername,age,gender,telephone1,telephone2,mr_no) values (patientnames,fathernames,ages,genders,telephone1s,telephone2s,mrnos);


ELSE

SIGNAL SQLSTATE '45000'
SET MESSAGE_TEXT = 'Patient MR Number Exist';

END IF;

 END




 /* Delete User  */


 CREATE Procedure deleteuserprocedure (IN delusername varchar(255), IN adminpassword varchar(255))


BEGIN

IF  EXISTS (select name from usertable where password = adminpassword) THEN


delete from usertable where username = delusername ;

ELSE

SIGNAL SQLSTATE '45000'
SET MESSAGE_TEXT = 'Password is incorrect or user not found';

END IF;

 END




 /* Add User */

CREATE Procedure adduserprocedures (IN newusername varchar(255), IN userpassword varchar(255), IN userrole varchar(255))


BEGIN

IF NOT EXISTS (select password from usertable where name = newusername) THEN

insert into usertable (name,password,role) values (newusername,userpassword,userrole);

ELSE

SIGNAL SQLSTATE '45000'
SET MESSAGE_TEXT = 'User already exist with this username';

END IF;

 END
